CREATE TABLE STUDENT (
USN VARCHAR(20),
SNAME CHAR(30),
DNO VARCHAR(10),
CNO VARCHAR(10),
SEM VARCHAR(5),
PRIMARY KEY(USN),
FOREIGN KEY(DNO) REFERENCES DEPARTMENT(DNO) ON DELETE CASCADE,
FOREIGN KEY(CNO) REFERENCES CLASSROOM(CNO) ON DELETE CASCADE);


CREATE TABLE DEPARTMENT(
DNO VARCHAR(10),
DNAME VARCHAR(30),
PRIMARY KEY(DNO));


CREATE TABLE CLASSROOM(
CNO VARCHAR(10),
NOOFROWS INT(11),
DNO VARCHAR(10),
PRIMARY KEY(CNO),
FOREIGN KEY(DNO) REFERENCES DEPARTMENT(DNO));

CREATE TABLE FACULTY(
DNO VARCHAR(10),
FID VARCHAR(30),
CNO VARCHAR(10),
FNAME CHAR(30),
PRIMARY KEY(FID),
FOREIGN KEY(DNO) REFERENCES DEPARTMENT(DNO) ON DELETE CASCADE,
FOREIGN KEY(CNO) REFERENCES CLASSROOM(CNO) ON DELETE CASCADE);


CREATE TABLE EXAM(
EID VARCHAR(30),
CNO VARCHAR(10),
ENAME VARCHAR(30),
EXAMDATE DATE,
PRIMARY KEY(EID),
FOREIGN KEY(CNO) REFERENCES CLASSROOM(CNO) ON DELETE CASCADE);
